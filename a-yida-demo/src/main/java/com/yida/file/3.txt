CREATE TABLE `oms_order` (
  `id` bigint(20) NOT NULL COMMENT '主键',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '更新时间',
  `pull_page_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '拉取日期',
  `meta_id` bigint(20) DEFAULT NULL COMMENT 'meta信息id',
  `platform_order_id` varchar(64) DEFAULT '' COMMENT '订单编号',
  `order_status` int(2) DEFAULT '0' COMMENT '订单状态：0.待处理 2.待审核 3.配货中  4.已发货 5.已完成 6.已作废',
  `order_weight` decimal(19,4) DEFAULT '0.0000' COMMENT '订单重量',
  `country_code` varchar(16) DEFAULT '' COMMENT '国家二字码',
  `paid_date` date DEFAULT NULL COMMENT '支付时间(日期）',
  `paid_time` datetime DEFAULT NULL COMMENT '订单付款时间',
  `sales_record_number` varchar(64) DEFAULT '' COMMENT '平台交易号',
  `platform_id` varchar(32) DEFAULT '' COMMENT '来源平台',
  `currency_id` varchar(32) DEFAULT '' COMMENT '币种',
  `country_name_en` varchar(64) DEFAULT '' COMMENT '国家英文名称',
  `country_name_cn` varchar(64) DEFAULT '' COMMENT '国家中文名称',
  `gateway` varchar(64) DEFAULT '' COMMENT '支付方式',
  `financial_status` varchar(32) DEFAULT '' COMMENT '付款状态',
  `express_time` datetime DEFAULT NULL COMMENT '发货时间',
  `email` varchar(128) DEFAULT '' COMMENT '邮件地址',
  `site_id` int(11) DEFAULT '0' COMMENT '站点id',
  `site_name` varchar(128) DEFAULT '' COMMENT '站点名称',
  `shipping_total_origin` decimal(19,4) DEFAULT '0.0000' COMMENT '原始运费收入',
  `item_total_origin` decimal(19,4) DEFAULT '0.0000' COMMENT '商品原始总售价',
  `total_price` decimal(19,4) DEFAULT '0.0000' COMMENT '总金额=商品原始总售价+原始运费收入+订单总小费-订单总折扣',
  `tax_total` decimal(20,4) DEFAULT '0.0000' COMMENT '订单总税费',
  `total_discounts` decimal(20,4) DEFAULT '0.0000' COMMENT '订单总折扣',
  `exchange_to_usd` decimal(19,4) DEFAULT '0.0000' COMMENT '当前币制转美元汇率',
  `total_tip_received` decimal(19,4) DEFAULT '0.0000' COMMENT '总小费',
  `shipping_cost` decimal(19,4) DEFAULT '0.0000' COMMENT '实际运费',
  `shipping_pre_cost` decimal(19,4) DEFAULT '0.0000' COMMENT '预估运费',
  `is_union` tinyint(2) DEFAULT '2' COMMENT '合并订单1.合并订单 2.非合并订单',
  `is_split` tinyint(2) DEFAULT '2' COMMENT '拆分订单1.拆分订单 2.非拆分订单',
  `is_resend` tinyint(2) DEFAULT '2' COMMENT '重发订单1.重发订单 2.非重发订单',
  `sync_delivery` tinyint(2) DEFAULT '0' COMMENT '交运 0 未同步，1：已同步 2,同步失败',
  `sync_wms` tinyint(2) DEFAULT '0' COMMENT '0 未同步，1：已同步 2,同步失败',
  `sync_wms_time` datetime DEFAULT NULL COMMENT '转wms时间',
  `has_goods` tinyint(2) DEFAULT '2' COMMENT '1.不缺货订单 2.缺货订单',
  `has_battery` tinyint(2) DEFAULT '2' COMMENT '1.带电订单 2.不带电订单',
  `has_magnetic` tinyint(2) DEFAULT '2' COMMENT '是否含磁1含2不含',
  `has_powder` tinyint(2) DEFAULT '2' COMMENT '是否含泡沫1含2不含',
  `has_tort` tinyint(2) DEFAULT '2' COMMENT '是否侵权1含2不含',
  `is_auto` tinyint(2) DEFAULT '1' COMMENT '是否自动(1-自动同步 2-手动信息，批量导入)',
  `is_domestic` tinyint(2) DEFAULT '1' COMMENT '1：国内仓，2：海外仓',
  `sync_mabang` tinyint(5) DEFAULT '0' COMMENT '是否同步至马帮 1是；0否',
  `exception_label_id` varchar(32) DEFAULT '' COMMENT '异常标签id',
  `order_type` tinyint(2) DEFAULT NULL COMMENT '订单类型',
  `platform_remark` text COMMENT '平台留言',
  `seller_message` text COMMENT '平台备注',
  `remark` text COMMENT '订单备注',
  `order_express_remark` text COMMENT '订单发货备注',
  `exception_label_name` varchar(255) DEFAULT NULL COMMENT '异常分类名称',
  `order_rule_id` bigint(20) DEFAULT NULL COMMENT '订单规则id',
  `order_rule_name` varchar(255) DEFAULT NULL COMMENT '订单规则名称',
  `declare_rule_id` bigint(20) DEFAULT NULL COMMENT '申报规则id',
  `declare_rule_name` varchar(255) DEFAULT NULL COMMENT '申报规则名称',
  `logistics_rule_id` bigint(20) DEFAULT NULL COMMENT '物流规则id',
  `logistics_rule_name` varchar(255) DEFAULT NULL COMMENT '物流规则名称',
  `is_single_product` tinyint(2) DEFAULT NULL COMMENT '是否单商品:0否1是',
  `is_multiple_warehouses` tinyint(2) DEFAULT NULL COMMENT '是否多仓库:0否1是',
  `exception_info` text COMMENT '交运/转wms异常信息',
  `has_paste` tinyint(2) DEFAULT '0' COMMENT '是否含膏体 1：是2：不是',
  `sync_time` datetime DEFAULT NULL COMMENT '同步时间',
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单明细表';
